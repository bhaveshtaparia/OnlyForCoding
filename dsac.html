How to create dynamicaliy array

#include <stdio.h>
#include <stdlib.h>
struct array{
int tsize;
int usize;
int *ptr;
};
void createarray(struct array *p,int u,int t ){
p->tsize=t;
p->usize=u;
p->ptr=(int*)malloc((p->tsize)*sizeof(int));
}
void putvalue(struct array *p){
for(int i=0;i<p->usize;i++){
int n;
printf("enter the value of index %d",i);
scanf("%d",&n);
(p->ptr)[i]=n;
}
}
void getvalue(struct array *p){
for(int i=0;i<p->usize;i++){
printf("element at index %d is %d",i,(p->ptr)[i]);
}
}
int main()
{struct array *p;
p=(struct array*)malloc(sizeof(struct array));
createarray(p,3,5);
putvalue(p);
getvalue(p);
return 0;
}
